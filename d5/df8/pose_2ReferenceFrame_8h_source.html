832  ReferenceFrame interpolate(const ReferenceFrame &other, ReferenceFrame parent, uint64_t time) const
833  {
834  double fraction = (time - timestamp()) / (double)(other.timestamp() - timestamp());
835 
836  Pose interp = origin();
837  const Pose &opose = other.origin();
838 
839  interp.frame(std::move(parent));

842  interp.y((fraction * (opose.y() - interp.y())) + interp.y());
843  interp.z((fraction * (opose.z() - interp.z())) + interp.z());
